Method: POST
URL: http://localhost:5000/api/staff/jobs/create
Headers: Authorization: Bearer <PASTE_STAFF_TOKEN_HERE>
Body: (Note the criteria: CGPA 7.0, Sem 6, CSE/IT)
json
 Show full code block 
{
  "company": {
    "name": "Google",
    "industry": "Tech"
  },
  "job": {
    "title": "Software Engineer",
    "description": "Backend Role",
    "deadline": "2025-12-31"
  },
  "criteria": {
    "min_cgpa": 7.0,
    "min_semester": 6,
    "departments": "CSE,IT"
  }
}
